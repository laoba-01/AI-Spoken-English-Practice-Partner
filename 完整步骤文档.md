// internal/handler/websocket.go
package handler

import (
    "net/http"
    "strings"
    "github.com/gin-gonic/gin"
    "github.com/gorilla/websocket"
    "github.com/yourname/english-tutor/internal/service"
)

var upgrader = websocket.Upgrader{
    CheckOrigin: func(r *http.Request) bool {
        return true // 生产环境配置允许的域名
    },
}

func WebSocketHandler(c *gin.Context) {
    conn, err := upgrader.Upgrade(c.Writer, c.Request, nil)
    if err != nil {
        return
    }
    defer conn.Close()

    conversationId := c.Param("id")
    scene, err := service.GetSceneByConversationId(conversationId)
    if err != nil {
        return
    }
    systemPrompt := service.GetSystemPromptByScene(scene)

    for {
        _, mp3Data, err := conn.ReadMessage()
        if err != nil {
            break
        }

        // 阿里云流式ASR识别
        text, err := service.ASRClient.StreamRecognize(mp3Data)
        if err != nil {
            conn.WriteJSON(map[string]interface{}{
                "type": "error",
                "message": "Sorry, I didn't catch that. Could you say it again?",
            })
            continue
        }

        // 实时推送识别结果给前端
        conn.WriteJSON(map[string]interface{}{
            "type": "asr_result",
            "text": text,
        })

        // 语音结束后调用豆包Pro流式生成
        if service.IsEndOfSpeech(mp3Data) {
            go func() {
                history, err := service.GetConversationHistory(conversationId)
                if err != nil {
                    return
                }

                stream, err := service.LLMClient.StreamChat(systemPrompt, history)
                if err != nil {
                    conn.WriteJSON(map[string]interface{}{
                        "type": "error",
                        "message": "System is busy, please try again later.",
                    })
                    return
                }

                var fullText strings.Builder
                for chunk := range stream {
                    fullText.WriteString(chunk)
                    conn.WriteJSON(map[string]interface{}{
                        "type": "llm_chunk",
                        "text": chunk,
                    })
                }

                // 完整回复生成后调用阿里云TTS
                mp3, err := service.TTSClient.TextToSpeech(fullText.String())
                if err != nil {
                    return
                }

                audioUrl, err := service.OSSClient.UploadMP3(mp3)
                if err != nil {
                    return
                }

                conn.WriteJSON(map[string]interface{}{
                    "type": "audio_result",
                    "audio_url": audioUrl,
                })

                // 保存对话记录
                service.SaveMessage(conversationId, "user", text, "")
                service.SaveMessage(conversationId, "assistant", fullText.String(), audioUrl)
            }()
        }
    }
}

CREATE TABLE user_conversations (
    id BIGINT PRIMARY KEY AUTO_INCREMENT COMMENT '会话ID',
    user_id BIGINT NOT NULL COMMENT '用户ID',
    scene VARCHAR(20) NOT NULL COMMENT '场景: daily/business/exam',
    title VARCHAR(100) DEFAULT NULL COMMENT '会话标题',
    created_at DATETIME DEFAULT CURRENT_TIMESTAMP COMMENT '创建时间',
    updated_at DATETIME DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP COMMENT '更新时间',
    INDEX idx_user_scene (user_id, scene) COMMENT '用户场景索引'
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COMMENT='对话会话表';
6.2 消息内容表
sql
CREATE TABLE user_messages (
    id BIGINT PRIMARY KEY AUTO_INCREMENT COMMENT '消息ID',
    conversation_id BIGINT NOT NULL COMMENT '会话ID',
    role VARCHAR(10) NOT NULL COMMENT '角色: user/assistant',
    content TEXT NOT NULL COMMENT '文字内容',
    audio_url VARCHAR(255) DEFAULT NULL COMMENT 'MP3语音URL',
    correction TEXT DEFAULT NULL COMMENT '纠错内容(JSON格式)',
    pronunciation_score TINYINT DEFAULT NULL COMMENT '发音评分0-100',
    created_at DATETIME DEFAULT CURRENT_TIMESTAMP COMMENT '创建时间',
    INDEX idx_conversation (conversation_id) COMMENT '会话索引'
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COMMENT='消息内容表';

10. 高考专项风险预案
10.1 内容安全误拦截
风险：白天考试时段，正常口语对话被误判为高考题
预案：
Prompt 中明确添加高考风控约束，降低误判率
捕获内容安全错误，自动重试 1 次
配置备用模型（通义千问 3.5），误拦截严重时切换
夜间无限制，可在夜间进行开发调试
10.2 API 限流
风险：高考期间大模型 API 限流严格
预案：
配置客户端限流，每个用户每分钟最多 10 次请求
配置服务端熔断，错误率 > 5% 时降级服务
提前充值足够的 API 额度，避免欠费停机
10.3 服务宕机
风险：服务器故障导致服务不可用
预案：
配置主备服务器，自动切换
开启阿里云 ECS 自动快照，定期备份数据
配置告警，服务宕机时立即通知开发人员
